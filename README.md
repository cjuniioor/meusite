# 1. Cria um novo projeto Next.js
npx create-next-app fluency-quiz-site

# 2. Entra na pasta do projeto
cd fluency-quiz-site

# 3. Instala TailwindCSS (para um visual mais lúdico e bonito)
npm install -D tailwindcss postcss autoprefixer

# 4. Inicializa o Tailwind
npx tailwindcss init -p

# 5. Substitui o conteúdo do tailwind.config.js para habilitar Tailwind no Next.js
# (feito manualmente, conforme instruções automaticamente mostradas)

# 6. Inicia o servidor de desenvolvimento
npm run dev
